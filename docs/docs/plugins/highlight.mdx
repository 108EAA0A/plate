### Installation

```bash npm2yarn
npm install @udecode/plate-highlight
```

### Usage

```ts live
() => {
  const ToolbarHighlight = () => {
    const editor = useStoreEditorRef(useEventEditorId('focus'));

    return (
      <ToolbarMark
        type={getPlatePluginType(editor, MARK_HIGHLIGHT)}
        icon={<Highlight />}
      />
    );
  };

  const plugins = [
    ...pluginsBasic,
    createHighlightPlugin(),
  ];

  return (
    <>
      <HeadingToolbar>
        <ToolbarHighlight />
      </HeadingToolbar>
      <Plate
        id="highlight"
        plugins={plugins}
        components={components}
        options={options}
        editableProps={editableProps}
        initialValue={initialValueHighlight}
      />
    </>
  );
}
```
