### Installation

```bash npm2yarn
npm install @udecode/plate-image
npm install @udecode/plate-image-ui
```

### Usage

```ts live
() => {
  const plugins = [
    ...pluginsBasic,
    createImagePlugin(),
    createSelectOnBackspacePlugin({ allow: [ELEMENT_IMAGE] }),
  ];

  return (
    <>
      <HeadingToolbar>
        <ToolbarImage icon={<Image />} />
      </HeadingToolbar>
      <Plate
        id="image"
        plugins={plugins}
        components={components}
        options={options}
        editableProps={editableProps}
        initialValue={initialValueImages}
      />
    </>
  );
}
```
