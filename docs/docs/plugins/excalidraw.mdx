### Installation

```bash npm2yarn
npm install @udecode/plate-excalidraw
```

### Usage

```ts live
() => {
  const plugins = [
    ...pluginsBasic,
    createExcalidrawPlugin(),
    createSelectOnBackspacePlugin({ allow: [ELEMENT_EXCALIDRAW] }),
  ];

  const styledComponents = {
    ...components,
    [ELEMENT_EXCALIDRAW]: ExcalidrawElement
  };

  return (
    <Plate
      id="excalidraw"
      plugins={plugins}
      components={styledComponents}
      options={options}
      editableProps={editableProps}
      initialValue={initialValueExcalidraw}
    />
  );
}
```
