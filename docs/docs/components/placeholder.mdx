### Installation

```bash npm2yarn
npm install @udecode/plate-placeholder
```

### Usage

```ts live
() => {
  const componentsPlaceholder = withStyledPlaceHolders(components);

  return (
    <>
      <HeadingToolbar>
        <ToolbarButtonsBasicElements />
      </HeadingToolbar>
      <Plate
        id="placeholder"
        plugins={pluginsBasic}
        components={componentsPlaceholder}
        options={options}
        editableProps={editableProps}
        initialValue={initialValuePlaceholder}
      />
    </>
  );
}
```
